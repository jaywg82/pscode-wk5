// Exercise 1: Calculate the Sum of Two Numbers

START
DECLARE variables num1, num2, sum.
READ values num1,num2.
CALCULATE sum==num1+num2
WRITE "Sum: ,sum"
STOP

// Exercise 2: Check if a Number is Even or Odd

START
DECLARE variables num
READ num
IF num%2==0 THEN
  WRITE "EVEN"
ELSE
  WRITE "ODD"
STOP
